import express, { type Request, Response, NextFunction } from "express";
import path from "path";
import cors from "cors";
import cookieParser from "cookie-parser";

import { registerRoutes } from "./routes";
import { setupVite, serveStatic, log } from "./vite";
import { setupUnifiedSession } from "./session-fix";
import { setupAuth } from "./auth";
import { fixRenderDirectories } from "./render-dir-fix";
import { logger, requestLogger, errorLogger } from "./logger";
import { configureForRender, isRenderEnvironment, getOptimizedDatabaseUrl } from "./render-integration";

// ğŸš€ Serverstart-Logging
logger.info("ğŸš€ Trading Journal Server startet...");

// ğŸ“ Verzeichnisse fÃ¼r Render vorbereiten
fixRenderDirectories();

// ğŸŒ Render-spezifische DB-Optimierung
if (isRenderEnvironment()) {
  try {
    process.env.DATABASE_URL = getOptimizedDatabaseUrl();
    logger.info("ğŸ“Š Datenbank-URL fÃ¼r Render optimiert (IPv4-Modus)");
  } catch (error) {
    logger.error("âŒ Fehler bei der Datenbank-URL-Optimierung", {
      error: error instanceof Error ? error.message : String(error)
    });
  }
}

const app = express();

// ğŸŒ Umgebung erkennen
const isRender = isRenderEnvironment();
const isReplit = !!process.env.REPL_ID || !!process.env.REPL_SLUG;
const isNetlify = process.env.NETLIFY === "true";
const isProduction = process.env.NODE_ENV === 'production';

logger.info("ğŸŒ Umgebung erkannt", {
  isRender,
  isReplit,
  isNetlify,
  isProduction,
  nodeEnv: process.env.NODE_ENV
});

// ğŸ› ï¸ Konfigurationen fÃ¼r Render aktivieren
configureForRender(app);

// ğŸ” Sessions initialisieren (wichtig vor setupAuth!)
setupUnifiedSession(app);

// ğŸŒ CORS
app.use(cors({
  origin: true,
  credentials: true,
  methods: ['GET', 'POST', 'PUT', 'DELETE', 'OPTIONS', 'PATCH'],
  allowedHeaders: ['Content-Type', 'Authorization', 'X-Requested-With', 'Accept', 'Origin', 'X-Client-Info']
}));

// ğŸª Cookies parsen
app.use(cookieParser());

// ğŸ§  Body-Parser fÃ¼r JSON & Formulardaten
app.use(express.json({ limit: '10mb' }));
app.use(express.urlencoded({ extended: false, limit: '10mb' }));

// ğŸ“‹ Logging
app.use(requestLogger);
app.use(errorLogger);

// ğŸ” Auth aktivieren
setupAuth(app);

// ğŸ–¼ï¸ Statische Dateien aus /public
app.use(express.static(path.join(process.cwd(), "public")));

// ğŸ“ Root-Redirect zu /auth
app.get("/", (req, res) => {
  logger.info("ğŸ“ Root-Pfad-Anfrage erkannt â€“ Weiterleitung zu /auth");
  return res.redirect("/auth");
});

// ğŸ§ª Erweiterte Logging-Middleware
app.use((req, res, next) => {
  const start = Date.now();
  const path = req.path;
  let capturedJsonResponse: any;

  if (path.startsWith("/api")) {
    console.log(`${new Date().toISOString()} - ${req.method} ${req.url}`);
    console.log(`Cookies: ${JSON.stringify(req.headers.cookie || 'none')}`);
    console.log(`Session ID: ${req.session?.id || 'keine Session'}`);
    console.log(`Authenticated: ${req.isAuthenticated()}`);
  }

  const originalResJson = res.json;
  res.json = function (bodyJson, ...args) {
    capturedJsonResponse = bodyJson;
    return originalResJson.apply(res, [bodyJson, ...args]);
  };

  res.on("finish", () => {
    const duration = Date.now() - start;
    if (path.startsWith("/api")) {
      let logLine = `${req.method} ${path} ${res.statusCode} in ${duration}ms`;
      if (capturedJsonResponse) {
        logLine += ` :: ${JSON.stringify(capturedJsonResponse)}`;
      }
      if (logLine.length > 80) {
        logLine = logLine.slice(0, 79) + "â€¦";
      }
      log(logLine);
    }
  });

  next();
});

(async () => {
  // ğŸ“¡ API-Routen initialisieren
  await registerRoutes(app);

  // âŒ Globale Fehlerbehandlung
  app.use((err: any, _req: Request, res: Response, _next: NextFunction) => {
    const status = err.status || err.statusCode || 500;
    const message = err.message || "Internal Server Error";

    if (_req.path.endsWith('.pdf')) {
      log(`PDF request: ${_req.path}`);
      return _next();
    }

    res.status(status).json({ message });
    throw err;
  });

  // âš™ï¸ Nur in Dev: Vite-Integration
  if (app.get("env") === "development") {
    await setupVite(app);
  } else {
    serveStatic(app);
  }

  // ğŸš€ Starte den Server auf Port 5000
  const port = 5000;
  app.listen(port, "0.0.0.0", () => {
    log(`ğŸš€ Server lÃ¤uft auf Port ${port}`);
  });
})();
